def plottv(i):
    
    xrange = []
    yrange = []

    for s, d in zip(list(plotdf.columns)[1:],list(plotdf.iloc[i,:])[1:]):
        xrange.append(float(s))
        yrange.append(float(d))

    plt.plot(xrange, yrange, label = plotdf.iloc[i,0])
